Overview

API ini digunakan untuk melihat produk, membuat transaksi, dan mengecek status. Semua endpoint menggunakan HTTPS. Balasan dalam format JSON.


kami juga menyediakan webhook untuk, informasi update status

GET /list_product

Lihat semua daftar produk tersedia.

Request GET 
https://panel.khfy-store.com/api_v2/list_product?api_key=API_KEY_ANDA

GET /trx

Buat transaksi baru.

Request GET
  • produk — kode produk yang valid (lihat List Produk)
  • tujuan — nomor tujuan
  • reff_id — opsional; jika kosong akan dibuat UUID acak
  • api_key — kunci API Anda
https://panel.khfy-store.com/api_v2/trx?produk=BPAL19&tujuan=087812345678&reff_id=123e4567-e89b-12d3-a456-426614174000&api_key=API_KEY_ANDA

Cek status transaksi by refid.

Request GET

Webhook

  1. Pasang URL webhook di Profile → Pengaturan.
  2. Setiap transaksi sukses/gagal akan dikirim ke URL tersebut.

Format panggilan

GET https://url-webhookkamu.com/webhook?message=<pesan_dari_server_khfy>

Payload status ada di message. Gunakan regex pada contoh di bawah.

Contoh Kode

Node.js (Express)


const express = require('express');

const app = express();
app.use(express.json({ limit: '1mb' }));
app.use(express.urlencoded({ extended: true }));

// Regex lebih fleksibel:
// - Tangkap reffid, trxid, produk, tujuan, status_text, keterangan
// - Bagian "Saldo ..." bersifat opsional
// - "result=\\d" opsional; kalau tak ada, kita derive dari status_text
const RX =
  /RC=(?<reffid>[a-f0-9-]+)\s+TrxID=(?<trxid>\d+)\s+(?<produk>[A-Z0-9]+)\.(?<tujuan>\d+)\s+(?<status_text>[A-Za-z]+)\s*(?<keterangan>.+?)(?:\s+Saldo[\s\S]*?)?(?:\bresult=(?<status_code>\d+))?\s*>?$/i;

app.all('/webhook', async (req, res) => {
  try {
    const message =
      (req.query && req.query.message) ||
      (typeof req.body?.message === 'string' ? req.body.message : null);

    if (!message) {
      console.log('[WEBHOOK] message kosong');
      return res.status(400).json({ ok: false, error: 'message kosong' });
    }

    const match = message.match(RX);
    if (!match || !match.groups) {
      console.log('[WEBHOOK] format tidak dikenali ->', message);
      return res.status(200).json({ ok: false, error: 'format tidak dikenali' });
    }

    const {
      trxid,
      reffid,
      produk,
      tujuan,
      status_text = '',
      status_code: statusCodeRaw,
    } = match.groups;

    // Keterangan: bersihkan trailing spasi
    const keterangan = (match.groups.keterangan || '').trim();

    // Normalisasi status_code:
    // 1) pakai result=... kalau ada
    // 2) jika tidak ada, derive dari status_text
    let status_code = null;
    if (statusCodeRaw != null) {
      status_code = Number(statusCodeRaw);
    } else {
      if (/sukses/i.test(status_text)) status_code = 0;
      else if (/gagal|batal/i.test(status_text)) status_code = 1;
    }

    // --- Console logs ringkas & jelas ---
    console.log('==== CALLBACK MASUK ====');
    console.log('RAW        :', message);
    console.log('reffid     :', reffid);
    console.log('trxid      :', trxid);
    console.log('produk     :', produk);
    console.log('tujuan     :', tujuan);
    console.log('status_txt :', status_text);
    console.log('status_code:', status_code);
    console.log('keterangan :', keterangan);
    console.log('=========================');

    // --- Blok aksi sesuai status_code (console.log dulu sesuai permintaan) ---
    if (Number(status_code) === 0) {
      console.log('[AKSI] Transaksi SUKSES -> jalankan flow sukses di sini');
      // TODO: taruh flow sukses (update DB, callback internal, dsb.)
    } else if (Number(status_code) === 1) {
      console.log('[AKSI] Transaksi GAGAL/DIBATALKAN -> jalankan flow gagal di sini');
      // TODO: taruh flow gagal
    } else {
      console.log('[AKSI] Status tidak dikenal / status_code null -> tandai untuk investigasi');
    }

    // Respons minimal agar pengirim tahu diterima
    return res.status(200).json({
      ok: true,
      parsed: { trxid, reffid, produk, tujuan, status_text, status_code, keterangan },
    });
  } catch (err) {
    console.log('[WEBHOOK][ERROR]', err?.message || err);
    return res.status(500).json({ ok: false, error: 'internal_error', detail: err?.message || String(err) });
  }
});

const PORT = process.env.PORT || 3300;
app.listen(PORT, () => console.log(`Webhook listening on :${PORT}`));

PHP


<?php
declare(strict_types=1);
header('Content-Type: application/json; charset=utf-8');

// Ambil message dari query, POST form, atau JSON body
$message = $_GET['message'] ?? null;

if ($message === null) {
    // Coba form-urlencoded
    if (isset($_POST['message']) && is_string($_POST['message'])) {
        $message = $_POST['message'];
    } else {
        // Coba JSON body
        $raw = file_get_contents('php://input');
        if ($raw !== false && strlen($raw) > 0) {
            $json = json_decode($raw, true);
            if (json_last_error() === JSON_ERROR_NONE && isset($json['message']) && is_string($json['message'])) {
                $message = $json['message'];
            }
        }
    }
}

if ($message === null || $message === '') {
    error_log('[WEBHOOK] message kosong');
    http_response_code(400);
    echo json_encode(['ok' => false, 'error' => 'message kosong'], JSON_UNESCAPED_UNICODE);
    exit;
}

// Regex fleksibel (mirip versi JS)
// - Tangkap reffid, trxid, produk, tujuan, status_text, keterangan
// - Segmen "Saldo ..." opsional
// - "result=\d" opsional; kalau tak ada, derive dari status_text
$pattern = '~RC=(?P<reffid>[a-f0-9-]+)\s+TrxID=(?P<trxid>\d+)\s+(?P<produk>[A-Z0-9]+)\.(?P<tujuan>\d+)\s+(?P<status_text>[A-Za-z]+)\s*(?P<keterangan>.+?)(?:\s+Saldo[\s\S]*?)?(?:\bresult=(?P<status_code>\d+))?\s*>?$~is';

if (!preg_match($pattern, $message, $m)) {
    error_log('[WEBHOOK] format tidak dikenali -> ' . $message);
    echo json_encode(['ok' => false, 'error' => 'format tidak dikenali'], JSON_UNESCAPED_UNICODE);
    exit;
}

// Ambil fields
$trxid       = $m['trxid']       ?? '';
$reffid      = $m['reffid']      ?? '';
$produk      = $m['produk']      ?? '';
$tujuan      = $m['tujuan']      ?? '';
$status_text = $m['status_text'] ?? '';
$ketRaw      = $m['keterangan']  ?? '';
$keterangan  = trim($ketRaw);

// Normalisasi status_code:
// 1) pakai result=... kalau ada
// 2) kalau tidak ada, derive dari status_text
$status_code = null;
if (isset($m['status_code']) && $m['status_code'] !== '') {
    $status_code = (int)$m['status_code'];
} else {
    if (preg_match('~sukses~i', $status_text)) {
        $status_code = 0;
    } elseif (preg_match('~gagal|batal~i', $status_text)) {
        $status_code = 1;
    }
}

// --- Logging ringkas ---
error_log('==== CALLBACK MASUK ====');
error_log('RAW        : ' . $message);
error_log('reffid     : ' . $reffid);
error_log('trxid      : ' . $trxid);
error_log('produk     : ' . $produk);
error_log('tujuan     : ' . $tujuan);
error_log('status_txt : ' . $status_text);
error_log('status_code: ' . var_export($status_code, true));
error_log('keterangan : ' . $keterangan);
error_log('=========================');

// --- Blok aksi sesuai status_code (console log via error_log) ---
if ($status_code === 0) {
    error_log('[AKSI] Transaksi SUKSES -> jalankan flow sukses di sini');
    // TODO: flow sukses (update DB, callback internal, dll)
} elseif ($status_code === 1) {
    error_log('[AKSI] Transaksi GAGAL/DIBATALKAN -> jalankan flow gagal di sini');
    // TODO: flow gagal
} else {
    error_log('[AKSI] Status tidak dikenal / status_code null -> tandai untuk investigasi');
}

// Response minimal
echo json_encode([
    'ok'     => true,
    'parsed' => [
        'trxid'       => $trxid,
        'reffid'      => $reffid,
        'produk'      => $produk,
        'tujuan'      => $tujuan,
        'status_text' => $status_text,
        'status_code' => $status_code,
        'keterangan'  => $keterangan,
    ],
], JSON_UNESCAPED_UNICODE);

Tips & Praktik Baik

  • Gunakan uuid unik untuk reff_id agar mudah dilacak.
  • Simpan mapping reff_id ↔ order internal di database Anda.
  • Timeout di sisi klien & retry sebaiknya idempotent (gunakan reff_id yang sama saat retry).
  • Validasi nomor tujuan sebelum memanggil endpoint /trx.
  • Whitelist IP server webhook Anda & verifikasi signature (opsional, jika nanti tersedia).

Quickstart

  1. Ambil api_key di Profile → Pengaturan.
  2. Cek produk: GET /list_product.
  3. Buat transaksi: GET /trx.
  4. Cek status: GET /history (pakai refid).
  5. Set webhook agar update status real-time diterima.
